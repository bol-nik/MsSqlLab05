возвращающий имена всех сотрудников, продававших товар в количестве ? 75 штук.
Select s.sp_name,
from sperson s
where sp_id in (Select sp_id from sale group by sp_id

Select s.sp_name,sum(qty) 
from Sperson s,Sale sa
where s.sp_id=(select sa.sp_id)
group by s.sp_name 
having sum(qty)<=75

Select s.sp_name,sum(qty) 
from Sperson s, Sale sa 
where sa.sp_id=s.sp_id 
group by s.sp_name 
having sum(qty)<=75

Select s.sp_name,sum(qty)
from Sperson s
right join Sale sa on sa.sp_id=s.sp_id
group by s.sp_name
having sum(qty)<=75
